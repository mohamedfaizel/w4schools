Angular TDS-FOrm : Submitting Form data

  • Create a new folder for server(open new visual studio code, out of the project)
  • npm init --yes
  • npm install --save express body-parser cors
    • express : its our web server
    • body-parse : its a middleware to handle form data.
    • cors : Its a package to make request to cross different ports
  • Create new file: server.js
  • npm serve
const express = require('express'); const bodyParser = require('body-parser'); const cors = require('cors'); const PORT = 3000; const app = express(); app.use(bodyParser.json()); app.use(cors()); app.get('/', function(req, res) { res.send('Hello from server'); }) app.post('/enroll', function(req, res) { console.log(req.body); res.status(200).send({'message' : 'Data Received'}); //(Error = )res.status(401).send({'message' : 'Data Received'}); }) app.listen(PORT, function() { console.log('Server running on localhost: '+ PORT) })

                        class Client{
                            constructor(name, balance){
                                this.name = name;
                                this.balance = balance;
                            }
                            membership(){
                                if(this.balance > 10000){
                                    return "Gold";
                                }else if(this.balance > 5000){
                                    return "Platinum";
                                }else{
                                    return "Normal";
                                }
                            }
                            withdraw(amount){
                                this.balance -= amount;
                            }
                            clientInfo(){
                                return `Name : ${this.name} balance : ${this.balance}`
                            }
                        }
                        const person1 = new Client("Lmj", 100000);
                        console.log(person1);
                        console.log(person1.clientInfo())

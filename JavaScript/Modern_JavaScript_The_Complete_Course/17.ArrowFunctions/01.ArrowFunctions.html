program 1

                    
                            //Example for genral function to arrow function
                            const shoppingCart = ["Album", "Shirt", "Gittar"];
                            const productQuantityLength = shoppingCart.map(function(product){
                                return product.length;
                            })
                            console.log(productQuantityLength)
                    
                            console.log("step 1 : remove function add arrow")
                            const productQuantityLength_step1 = shoppingCart.map((product) => {
                                return product.length;
                            })
                            console.log(productQuantityLength_step1)
                    
                            console.log("step 2 : function body is one line so remove the {} and return also must remove")
                            const productQuantityLength_step2 = shoppingCart.map((product) => product.length )
                            console.log(productQuantityLength_step2)
                    
                            console.log("step 3 : arrow function have only one parameter, so remove the ()")
                            const productQuantityLength_step3 = shoppingCart.map(product => product.length )
                            console.log(productQuantityLength_step3)

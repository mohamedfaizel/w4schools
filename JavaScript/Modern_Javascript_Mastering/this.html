this - Global object

this indicates the closest parent object

console.log(this); // indicates window object function test(){ console.log(this) // here the closest parent object is window } var greet = { name : "Lmj" } test.call(greet); // test function console print the greet object because of closest parent object is greet; Window {postMessage: ƒ, blur: ƒ, focus: ƒ, close: ƒ, parent: Window, …} {name: "Lmj"}

this - inside object

nested object this indicates the which one is closest parent object

var person = { fname : "syed", lname : "abbas", fullname : function (){ console.log(this.fname +' '+this.lname) }, owner: { fname : "Lmj", lname : "faizel", fullname : function (){ console.log(this.fname +' '+this.lname) } } } person.fullname(); // syed abbas person.owner.fullname() // Lmj faizel

this - in new keyword

function car(model){ this.model = model } var mycar = new car("Escape"); console.log(mycar); // car{model:"Escape"}
